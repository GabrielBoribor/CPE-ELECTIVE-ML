import cv2
import numpy as np
import matplotlib.pyplot as plt
from PIL import Image
import os

image_path = r'C:\Users\Borvs\Desktop\test\apple.jfif' 
if not os.path.exists(image_path):
    raise FileNotFoundError(f"The file '{image_path}' does not exist.")

# Open the image using PIL and convert to numpy array
image = Image.open(image_path)
image_np = np.array(image)

# Convert the image to grayscale (para sa edge detection)
gray_image = cv2.cvtColor(image_np, cv2.COLOR_BGR2GRAY) #machine learning tol

# Define convolution kernels
edge_detection_kernel = np.array([[-1, -1, -1],
                                  [-1, 8, -1],
                                  [-1, -1, -1]])

blurring_kernel = np.ones((5, 5), dtype=np.float32) / 25

# Apply 2D convolution for edge detection
edge_detected_image = cv2.filter2D(gray_image, -1, edge_detection_kernel)

# Apply 2D convolution for blurring
blurred_image = cv2.filter2D(image_np, -1, blurring_kernel)

# Plotting the images
plt.figure(figsize=(15, 10))

# Original Image
plt.subplot(1, 3, 1)
plt.imshow(image_np)
plt.title('Original Image')
plt.axis('off')

# Edge Detected Image
plt.subplot(1, 3, 2)
plt.imshow(edge_detected_image, cmap='gray')
plt.title('Edge Detection')
plt.axis('off')

# Blurred Image
plt.subplot(1, 3, 3)
plt.imshow(cv2.cvtColor(blurred_image, cv2.COLOR_BGR2RGB))
plt.title('Blurred Image')
plt.axis('off')

plt.tight_layout()
plt.show()
